163  private void messageHandlerSocket(byte bRead, byte prevByte) {
164 
165  byte[,] mainCommands = { /* {command,databytes} */
166  {128,0}
167  ,{129,1}
168  ,{130,0}
169  ,{131,0}
170  ,{132,0}
171  ,{133,0}
172  ,{134,0}

176  ,{145,4}
177  ,{138,1}

180  ,{139,3}
181  ,{140,2}
182  ,{142,1}
183  ,{148,1}
184  ,{149,1}
185  ,{150,1}
186  };
187 
188  byte[] storeTemp = {0,0};
189 
190  try {
191 
192  // decide: is this a mainCommand or a databyte
193  if (command == 0) {
194 
195  for (int i = 0; i < mainCommands.Length / 2; i++) {
196  if (mainCommands[i, 0] == bRead) {
197  command = mainCommands[i, 0];
198  byteCount = mainCommands[i, 1];
199  log(String.Format("Main command received: {0}", bRead), logTags.program);
200  break; // no need to keep looping
201  }
202  }
203 
204  } else {
205 
206  dataBytes.Add(bRead);
207  log(String.Format("Databyte received: {0}", bRead), logTags.program);
208 
209  // dynamic byte count
210  if (command == 140 && dataBytes.Count == 2) {
211  byteCount = (byte)((dataBytes[1]*2) + (byte)2);
212  } else if ((command == 148 || command == 149 ) && dataBytes.Count == 1) {
213  byteCount = (byte)(dataBytes[0] + (byte)1);
214  }
215 
216  }
217 
218  if (dataBytes.Count == byteCount) {
219 
220  switch (command) {
221  case 128: roomba.start(); break;
222  case 130: // catch this one, set to safe.
223  case 131: roomba.safe(); break;
224  case 132: roomba.full(); break;
225  case 133: roomba.power(); break;
226  case 134: roomba.spot(); break;
227  case 135: roomba.clean(); break;
228  case 136: roomba.max(); break;
229  case 137: roomba.drive(dataBytes[0], dataBytes[1], dataBytes[2], dataBytes[3]); break;
230  case 145: roomba.driveDirect(dataBytes[0], dataBytes[1], dataBytes[2], dataBytes[3]); break;
231  case 138: roomba.motors(dataBytes[0]); break;
232  case 144: roomba.pwmMotors(dataBytes[0], dataBytes[1], dataBytes[2]); break;
233  case 146: roomba.drivePwm(dataBytes[0], dataBytes[1], dataBytes[2], dataBytes[3]); break;
234  case 139: roomba.leds(dataBytes[0], dataBytes[1], dataBytes[2]); break;
235  case 140: storeTemp[0] = dataBytes[0];
236  storeTemp[1] = dataBytes[1];
237  dataBytes.RemoveRange(0, 2);
238  roomba.song(storeTemp[0], storeTemp[1], dataBytes.ToArray());
239  break;
240  case 142: roomba.getSensor(dataBytes[0]); break;
241  case 148: storeTemp[0] = dataBytes[0];
242  dataBytes.RemoveRange(0, 1);
243  roomba.startStream(storeTemp[0], dataBytes.ToArray());
244  break;
245  case 149: storeTemp[0] = dataBytes[0];
246  dataBytes.RemoveRange(0, 1);

248  break;
