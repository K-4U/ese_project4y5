217 
218  byte[,] mainCommands = { /* {command,databytes} */
219  {128,0}

244

251  byteCount = mainCommands[i, 1];
252  log(String.Format("Main command received: {0}", bRead), logTags.program);
253  break; // no need to keep looping
254  }
255  }
256 
257  } else {
258

260  log(String.Format("Databyte received: {0}", bRead), logTags.program);
261 
262  // dynamic byte count
263  if (command == 140 && dataBytes.Count == 2) {
264  byteCount = (byte)((dataBytes[1]*2) + (byte)2);
265  } else if ((command == 148 || command == 149 ) && dataBytes.Count == 1) {
266  byteCount = (byte)(dataBytes[0] + (byte)1);
267  }
268 
269  }
270 
271  if (dataBytes.Count == byteCount) {
272 
273  switch (command) {
274  case 128: roomba.start(); break;
275  case 130: // catch this one, set to safe.
276  case 131: roomba.safe(); break;
277  case 132: roomba.full(); break;
278  case 133: roomba.power(); break;
279  case 134: roomba.spot(); break;
280  case 135: roomba.clean(); break;
281  case 136: roomba.max(); break;
282  case 137: roomba.drive(dataBytes[0], dataBytes[1], dataBytes[2], dataBytes[3]); break;
283  case 145: roomba.driveDirect(dataBytes[0], dataBytes[1], dataBytes[2], dataBytes[3]); break;
284  case 138: roomba.motors(dataBytes[0]); break;
285  case 144: roomba.pwmMotors(dataBytes[0], dataBytes[1], dataBytes[2]); break;
286  case 146: roomba.drivePwm(dataBytes[0], dataBytes[1], dataBytes[2], dataBytes[3]); break;
